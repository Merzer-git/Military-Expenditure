import streamlit as st
import pandas as pd
import plotly.express as px

df = pd.read_parquet("data/datos_militares_final.parquet")

fig = px.scatter(
    df,
    x= 'Share_of_GDP',
    y= 'Spending_B',
    color= 'Historical_Era',
    hover_name= 'Country',
    log_y= True,
    title= "La evolucion del esfuerzo militar"
)

st.plotly_chart(fig)
